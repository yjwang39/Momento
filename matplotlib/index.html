from cartopy.mpl.ticker import LatitudeFormatter, LongitudeFormatter
import xarray as xr
import numpy as np
import matplotlib.pyplot as plt
import cmaps as cmap
import cartopy.feature as cfeat
import cartopy.crs as ccrs
plt.rcParams['font.sans-serif']=['Helvetica']

proj = ccrs.PlateCarree(central_longitude=0)
extent=[-180, 180, -90, 90]

def create_map_single(ax):
    ax.add_feature(cfeat.COASTLINE.with_scale('110m'),lw=0.5)
    ax.add_feature(cfeat.LAKES.with_scale('110m'))
    # ax.add_feature(cfeat.RIVERS.with_scale('110m'),lw=0.5)
    gl = ax.gridlines(crs=proj,draw_labels=False,linewidth=0.5,color='k',alpha=0.5,linestyle='--')
    gl.top_labels = gl.right_labels = gl.left_labels = gl.bottom_labels = False
    ax.set_xticks(np.arange(extent[0], extent[1]+1, 30))
    ax.set_yticks(np.arange(extent[2], extent[3]+1, 30))
    ax.xaxis.set_major_formatter(LongitudeFormatter())
    ax.xaxis.set_minor_locator(plt.MultipleLocator(10))
    ax.yaxis.set_major_formatter(LatitudeFormatter())
    ax.yaxis.set_minor_locator(plt.MultipleLocator(10))
    ax.tick_params(axis='both',labelsize=6,direction='out',right=True,top=True)
    ax.set_extent(extent,crs=proj)
    return ax
def create_map_mult(ax):
    ax.add_feature(cfeat.COASTLINE.with_scale('110m'),lw=0.5)
    ax.add_feature(cfeat.LAKES.with_scale('110m'))
    # ax.add_feature(cfeat.RIVERS.with_scale('110m'),lw=0.5)
    # gl = ax.gridlines(crs=proj,draw_labels=False,linewidth=0.5,color='k',alpha=0.5,linestyle='--')
    # gl.top_labels = gl.right_labels = gl.left_labels = gl.bottom_labels = False
    ax.set_xticks(np.arange(extent[0], extent[1]+1, 60))
    ax.set_yticks(np.arange(extent[2], extent[3]+1, 30))
    ax.xaxis.set_major_formatter(LongitudeFormatter())
    ax.xaxis.set_minor_locator(plt.MultipleLocator(10))
    ax.yaxis.set_major_formatter(LatitudeFormatter())
    ax.yaxis.set_minor_locator(plt.MultipleLocator(10))
    ax.tick_params(axis='both',width=0.5,pad=2,labelsize=6,direction='out',right=True,top=True)
    ax.set_extent(extent,crs=proj)
    return ax
